% curva equilibrio: y = alpha*x/(1+(alpha-1)*x)
% retta alta: yR = E/(E+1)*x + xD/(E+1)
\pgfmathsetmacro{\mRMT}{\EopMT/(\EopMT+1)}
\pgfmathsetmacro{\bRMT}{\xDMT/(\EopMT+1)}
\pgfmathsetmacro{\yPPMT}{\mRMT*\zFMT+\bRMT}

% retta bassa: passa per (xB,xB) e P=(zF,yP)
\pgfmathsetmacro{\mSMT}{(\yPPMT-\xBMT)/(\zFMT-\xBMT)}
\pgfmathsetmacro{\bSMT}{\xBMT-\mSMT*\xBMT}

% pinch per q-line verticale: Ppinch = (zF, y*(zF))
\pgfmathsetmacro{\yPinchMT}{\alphaMT*\zFMT/(1+(\alphaMT-1)*\zFMT)}
\pgfmathsetmacro{\mMinMT}{(\yPinchMT-\xDMT)/(\zFMT-\xDMT)}
\pgfmathsetmacro{\bMinMT}{\xDMT-\mMinMT*\xDMT}
